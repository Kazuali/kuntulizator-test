const CONFIG = {

    r3: { sheetName: '3-тур', title: '3 тур', deadline: '2026-06-24T22:00:00+03:00', order: 3 }

function doGet(e) {
  const callback = String((e && e.parameter && e.parameter.callback) || 'callback');
  try {
    const action = String((e && e.parameter && e.parameter.action) || '');
    const payloadText = String((e && e.parameter && e.parameter.payload) || '{}');
    const payload = JSON.parse(payloadText || '{}');

    if (action === 'submitPredictions') return jsonp_(callback, submitPredictions_(payload));
    if (action === 'getMyPredictions') return jsonp_(callback, getMyPredictions_(payload));

    return jsonp_(callback, { ok: false, message: 'Неизвестное действие.' });
  } catch (err) {
    return jsonp_(callback, { ok: false, message: err && err.message ? err.message : String(err) });
  }
}

function submitPredictions_(payload) {
  const roundKey = clean_(payload.roundKey || 'r1');
  const round = getRound_(roundKey);
  const now = new Date();
  const deadline = new Date(round.deadline);
  if (now.getTime() >= deadline.getTime()) throw new Error('Приём прогнозов на этот тур уже закрыт.');

  const participantName = clean_(payload.participantName);
  const pin = clean_(payload.pin);
  const predictions = Array.isArray(payload.predictions) ? payload.predictions : [];
  validateParticipant_(participantName, pin);

  const ctx = getSheetContext_(round);
  const participantRowIndex = findParticipantRow_(ctx.values, participantName, ctx.headerRowIndex, ctx.resultRowIndex);
  if (participantRowIndex === -1) throw new Error('Не найдена строка участника в таблице: ' + participantName);

  const eligibleColumns = getEligibleMatchColumns_(ctx.matchColumns, participantName, roundKey);
  if (predictions.length !== eligibleColumns.length) {
    throw new Error('Нужно заполнить все доступные матчи тура целиком. Матчей: ' + eligibleColumns.length + '. Получено прогнозов: ' + predictions.length + '.');
  }

  const eligibleByColumn = {};
  eligibleColumns.forEach(function (m) { eligibleByColumn[m.sheetColumn] = m; });

  const byColumn = {};
  predictions.forEach(function (p) {
    const sheetColumn = Number(p.sheetColumn);
    const home = Number(p.home);
    const away = Number(p.away);
    if (!eligibleByColumn[sheetColumn]) throw new Error('Этот матч недоступен для участника: колонка ' + sheetColumn);
    if (!Number.isInteger(home) || !Number.isInteger(away) || home < 0 || away < 0 || home > 20 || away > 20) {
      throw new Error('Счёт должен быть целым числом от 0 до 20.');
    }
    byColumn[sheetColumn] = home + '-' + away;
  });

  eligibleColumns.forEach(function (m) {
    if (!Object.prototype.hasOwnProperty.call(byColumn, m.sheetColumn)) throw new Error('Не заполнен матч: ' + m.title);
  });

  const rowNumber = participantRowIndex + 1;
  eligibleColumns.forEach(function (m) {
    const cell = ctx.sheet.getRange(rowNumber, m.sheetColumn);
    cell.setNumberFormat('@');
    cell.setValue(byColumn[m.sheetColumn]);
  });

  return { ok: true, message: 'Прогнозы сохранены: ' + participantName + ', ' + round.title + '. До дедлайна можно отправить новую версию.' };
}

function getMyPredictions_(payload) {
  const roundKey = clean_(payload.roundKey || 'r1');
  const round = getRound_(roundKey);
  const participantName = clean_(payload.participantName);
  const pin = clean_(payload.pin);
  validateParticipant_(participantName, pin);

  const ctx = getSheetContext_(round);
  const participantRowIndex = findParticipantRow_(ctx.values, participantName, ctx.headerRowIndex, ctx.resultRowIndex);
  if (participantRowIndex === -1) throw new Error('Не найдена строка участника в таблице: ' + participantName);

  const row = ctx.values[participantRowIndex] || [];
  const eligibleColumns = getEligibleMatchColumns_(ctx.matchColumns, participantName, roundKey);
  const predictions = eligibleColumns.map(function (m) {
    const score = parseScore_(row[m.zeroBasedColumn]);
    return { sheetColumn: m.sheetColumn, matchTitle: m.title, home: score ? score.home : null, away: score ? score.away : null };
  });

  return { ok: true, participantName: participantName, roundTitle: round.title, predictions: predictions };
}

function getRound_(roundKey) {
  const key = clean_(roundKey || 'r1');
  const round = CONFIG.rounds[key];
  if (!round) throw new Error('Неизвестный тур: ' + key);
  return round;
}

function getSheetContext_(round) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(round.sheetName);
  if (!sheet) throw new Error('Не найден лист: ' + round.sheetName);

  const values = sheet.getDataRange().getValues();
  const headerRowIndex = findHeaderRow_(values);
  const resultRowIndex = findRowByFirstCellContains_(values, CONFIG.resultLabel, headerRowIndex + 1);

  if (headerRowIndex === -1) throw new Error('Не найдена строка заголовков: Участник / Матч');
  if (resultRowIndex === -1) throw new Error('Не найдена строка результата: ' + CONFIG.resultLabel);
  if (resultRowIndex <= headerRowIndex) throw new Error('Строка результата должна быть ниже участников.');

  const headerRow = values[headerRowIndex];
  const matchColumns = [];
  for (let col = 1; col < headerRow.length; col++) {
    const title = clean_(headerRow[col]);
    if (title) matchColumns.push({ sheetColumn: col + 1, zeroBasedColumn: col, title: title });
  }
  if (!matchColumns.length) throw new Error('Не найдены матчи в строке заголовков.');
  return { ss: ss, sheet: sheet, values: values, headerRowIndex: headerRowIndex, resultRowIndex: resultRowIndex, matchColumns: matchColumns };
}

function getEligibleMatchColumns_(matchColumns, participantName, roundKey) {
  const rule = CONFIG.participantRules[participantName] || {};
  const activeRoundKey = clean_(rule.activeFromRoundKey || 'r1');
  const currentRound = getRound_(roundKey);
  const activeRound = getRound_(activeRoundKey);

  if ((currentRound.order || 999) < (activeRound.order || 999)) return [];
  if ((currentRound.order || 999) > (activeRound.order || 999)) return matchColumns;

  const activeFromMatchTitle = clean_(rule.activeFromMatchTitle || '');
  if (!activeFromMatchTitle) return matchColumns;

  const startIndex = matchColumns.findIndex(function (m) { return norm_(m.title) === norm_(activeFromMatchTitle); });
  if (startIndex === -1) return matchColumns;
  return matchColumns.slice(startIndex);
}

function validateParticipant_(participantName, pin) {
  if (!participantName) throw new Error('Не выбран участник.');
  if (!Object.prototype.hasOwnProperty.call(CONFIG.pins, participantName)) throw new Error('Участник не найден.');
  if (CONFIG.pins[participantName] !== pin) throw new Error('Неверный PIN-код.');
}

function jsonp_(callback, obj) {
  const safeCallback = String(callback).replace(/[^a-zA-Z0-9_.$]/g, '');
  const body = safeCallback + '(' + JSON.stringify(obj) + ');';
  return ContentService.createTextOutput(body).setMimeType(ContentService.MimeType.JAVASCRIPT);
}
function clean_(value) { return String(value == null ? '' : value).replace(/\s+/g, ' ').trim(); }
function norm_(value) { return clean_(value).toLowerCase().replace(/[–—-]/g, '-').replace(/\s*-\s*/g, '-'); }
function parseScore_(value) {
  const raw = clean_(value);
  if (!raw) return null;
  const normalized = raw.replace(/[—–]/g, '-').replace(/[.:,]/g, '-').replace(/\s+/g, '');
  const match = normalized.match(/^(\d{1,2})-(\d{1,2})$/);
  if (!match) return null;
  return { home: Number(match[1]), away: Number(match[2]) };
}
function findHeaderRow_(values) {
  for (let i = 0; i < values.length; i++) {
    const first = clean_(values[i][0]).toLowerCase();
    if (!first) continue;
    for (let j = 0; j < CONFIG.headerLabels.length; j++) {
      const target = clean_(CONFIG.headerLabels[j]).toLowerCase();
      if (first.indexOf(target) !== -1 || target.indexOf(first) !== -1) return i;
    }
    if (first.indexOf('участник') !== -1 && first.indexOf('матч') !== -1) return i;
  }
  return -1;
}
function findRowByFirstCellContains_(values, label, startIndex) {
  const target = clean_(label).toLowerCase();
  for (let i = startIndex || 0; i < values.length; i++) {
    const first = clean_(values[i][0]).toLowerCase();
    if (first.indexOf(target) !== -1) return i;
  }
  return -1;
}
function findParticipantRow_(values, participantName, headerRowIndex, resultRowIndex) {
  const target = clean_(participantName).toLowerCase();
  for (let i = headerRowIndex + 1; i < resultRowIndex; i++) {
    const first = clean_(values[i][0]).toLowerCase();
    if (first === target) return i;
  }
  return -1;
}
